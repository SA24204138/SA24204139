gibbs_sampler <- function(iterations, n, a, b) {
  x <- numeric(iterations)
  y <- numeric(iterations)
  y[1] <- 0.5  
  for (i in 2:iterations) {
    x[i] <- rbinom(1, size = n, prob = y[i - 1])
    y[i] <- rbeta(1, shape1 = x[i] + a, shape2 = n - x[i] + b)
  }
  
  return(data.frame(x = x, y = y))
}
set.seed(123)
